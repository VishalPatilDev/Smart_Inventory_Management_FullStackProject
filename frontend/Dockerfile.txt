# smart_inventory_management/frontend/Dockerfile
# ── Stage 1: Build the React app using Node ───────────────────────────────────
FROM node:18-alpine AS build

WORKDIR /app

# Copy package files first for dependency caching
COPY package.json package-lock.json ./
RUN npm ci --silent

# Copy rest of source and build
COPY . .
RUN npm run build
# Output goes to /app/dist (Vite default)

# ── Stage 2: Serve with Nginx ─────────────────────────────────────────────────
# nginx:alpine is only ~25MB — perfect for static files
FROM nginx:alpine

# Remove default Nginx page
RUN rm -rf /usr/share/nginx/html/*

# Copy built React files from stage 1
COPY --from=build /app/dist /usr/share/nginx/html

# Copy our custom Nginx config (handles React Router + API proxy)
COPY nginx.conf /etc/nginx/conf.d/default.conf

EXPOSE 80

CMD ["nginx", "-g", "daemon off;"]
